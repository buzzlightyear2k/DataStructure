class Solution {
public:
    bool isPalindrome(long long x) {
        long long y=x;
        if(x<0) return false;    
        if(x%10==0 && x!=0) return false;
        long long temp=0;
        while(x!=0){
            temp=temp*10+x%10;
            x/=10;
        }
        if(temp==y) return true;
        else return false;
    }
};
